char cor;

int red = 3;
int green = 4;
int yellow = 5;

void setup() {
Serial.begin(9600);

pinMode(red, OUTPUT);
pinMode(green, OUTPUT);
pinMode(yellow,OUTPUT);

}
void loop() {
while (Serial.available() > 0) {
cor = Serial.read();
  
if (cor == 'r' || cor =='R'){
digitalWrite(red, HIGH);
digitalWrite(green, LOW);
digitalWrite(yellow, LOW);
}

else if (cor == 'g' || cor =='G'){
digitalWrite(red, LOW);
digitalWrite(green, HIGH);
digitalWrite(yellow, LOW);
}

else if (cor == 'y' || cor =='Y'){
digitalWrite(red, LOW);
digitalWrite(green, LOW);
digitalWrite(yellow, HIGH);
}
Serial.println(cor);
  }
}
